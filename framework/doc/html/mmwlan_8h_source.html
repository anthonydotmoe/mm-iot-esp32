264static inline const struct mmwlan_s1g_channel_list *
265mmwlan_lookup_regulatory_domain(const struct mmwlan_regulatory_db *db, const char *country_code)
266{
267 unsigned ii;
268
269 if (db == NULL)
270 {
271 return NULL;
272 }
273
274 for (ii = 0; ii < db->num_domains; ii++)
275 {
276 const struct mmwlan_s1g_channel_list *channel_list = db->domains[ii];
277 if (channel_list->country_code[0] == country_code[0] &&
278 channel_list->country_code[1] == country_code[1])
279 {
280 return channel_list;
281 }
282 }
283 return NULL;
284}

1986static inline enum mmwlan_status mmwlan_tx_tid(const uint8_t *data, unsigned len, uint8_t tid)
1987{
1988 enum mmwlan_status status;
1989 struct mmpkt *pkt;
1990 struct mmpktview *pktview;
1991 struct mmwlan_tx_metadata metadata = MMWLAN_TX_METADATA_INIT;
1992
1993 status = mmwlan_tx_wait_until_ready(MMWLAN_TX_DEFAULT_TIMEOUT_MS);
1994 if (status != MMWLAN_SUCCESS)
1995 {
1996 return status;
1997 }
1998
1999 pkt = mmwlan_alloc_mmpkt_for_tx(len, tid);
2000 if (pkt == NULL)
2001 {
2002 return MMWLAN_NO_MEM;
2003 }
2004
2005 pktview = mmpkt_open(pkt);
2006 mmpkt_append_data(pktview, data, len);
2007 mmpkt_close(&pktview);
2008
2009 metadata.tid = tid;
2010 return mmwlan_tx_pkt(pkt, &metadata);
2011}
