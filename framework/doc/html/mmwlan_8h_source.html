264static inline const struct mmwlan_s1g_channel_list *
265mmwlan_lookup_regulatory_domain(const struct mmwlan_regulatory_db *db, const char *country_code)
266{
267 unsigned ii;
268
269 if (db == NULL)
270 {
271 return NULL;
272 }
273
274 for (ii = 0; ii < db->num_domains; ii++)
275 {
276 const struct mmwlan_s1g_channel_list *channel_list = db->domains[ii];
277 if (channel_list->country_code[0] == country_code[0] &&
278 channel_list->country_code[1] == country_code[1])
279 {
280 return channel_list;
281 }
282 }
283 return NULL;
284}

2024static inline enum mmwlan_status mmwlan_tx_tid(const uint8_t *data, unsigned len, uint8_t tid)
2025{
2026 enum mmwlan_status status;
2027 struct mmpkt *pkt;
2028 struct mmpktview *pktview;
2029 struct mmwlan_tx_metadata metadata = MMWLAN_TX_METADATA_INIT;
2030
2031 status = mmwlan_tx_wait_until_ready(MMWLAN_TX_DEFAULT_TIMEOUT_MS);
2032 if (status != MMWLAN_SUCCESS)
2033 {
2034 return status;
2035 }
2036
2037 pkt = mmwlan_alloc_mmpkt_for_tx(len, tid);
2038 if (pkt == NULL)
2039 {
2040 return MMWLAN_NO_MEM;
2041 }
2042
2043 pktview = mmpkt_open(pkt);
2044 mmpkt_append_data(pktview, data, len);
2045 mmpkt_close(&pktview);
2046
2047 metadata.tid = tid;
2048 return mmwlan_tx_pkt(pkt, &metadata);
2049}
