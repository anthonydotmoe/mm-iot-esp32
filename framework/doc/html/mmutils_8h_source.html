123static inline char mm_nibble_to_hex_char(uint8_t nibble)
124{
125 nibble &= 0x0f;
126 if (nibble < 0x0a)
127 {
128 return '0' + nibble;
129 }
130 else
131 {
132 return 'A' + nibble - 0x0a;
133 }
134}
